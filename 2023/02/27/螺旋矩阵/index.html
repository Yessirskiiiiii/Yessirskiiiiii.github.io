顺时针打印矩阵（剑指 Offer 29）

  • 问题描述：输入一个矩阵，按照从外向里以顺时针的顺序依次打印出每一个数字。

  • class Solution {
    public int[] spiralOrder(int[][] matrix) {
        // 非空判断
        if (matrix == null || matrix.length == 0 || matrix[0].length == 0) {
            return new int[0];
        }
        int m = matrix.length, n = matrix[0].length;
        int target = m * n, index = 0;
        int[] res = new int[target];
        // 定义上下左右四个边界
        int top = 0, right = n - 1, bottom = m - 1, left = 0;
        while (index < target) {
            // 左 → 右 执行完上边界向下移动 1
            for (int i = left; i <= right && index < target; i++) {
                res[index++] = matrix[top][i];
            }
            top++;
            // 上 → 下 执行完右边界向左移动 1
            for (int i = top; i <= bottom && index < target; i++) {
                res[index++] = matrix[i][right];
            }
            right--;
            // 右 → 左 执行完下边界向上移动 1
            for (int i = right; i >= left && index < target; i--) {
                res[index++] = matrix[bottom][i];
            }
            bottom--;
            // 下 → 上 执行完左边界向右移动 1
            for (int i = bottom; i >= top && index < target; i--) {
                res[index++] = matrix[i][left];
            }
            left++;
        }
        return res;
    }
}
